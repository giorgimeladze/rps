2. bundle exec figaro install
3. add GAME_BASE_URI: 'https://5eddt4q9dk.execute-api.us-east-1.amazonaws.com'
GAME_PATH: '/rps-stage/throw'
into application.yml as an ENV variables
4. yarn install
5. yarn run build
6. rails s
